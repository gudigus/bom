  CREATE OR REPLACE FUNCTION "BOM"."GET_USER_INFO" 
(p_code in number ,              -- 1) parameter에 따른 Return: user_info.uimage[1], user_info.unickname[2],user_info.uatid[3]
 p_u_code in board.bbcode%type)  -- 2) 사용자 정보 
 Return VARCHAR2
    
IS  
    v_uimage     user_info.uimage%type;  -- 프로그램 실행시 나타나는 변수
    v_unickname  user_info.unickname%type;  -- 프로그램 실행시 나타나는 변수
    v_uatid      user_info.uatid%type;  -- 프로그램 실행시 나타나는 변수

Begin   
   --2) 사번을 가지고 user_info.uimage[1], user_info.unickname[2],user_info.uatid[3]를 구함
   Select uimage     ,  unickname      , uatid
   Into   v_uimage   ,  v_unickname    , v_uatid
   From   user_info
   Where  ucode=p_u_code;
 
   --3) p_code를 가지고 Return
   IF     p_code = 1        THEN
          Return v_uimage;
   
   ELSIF  p_code = 2          THEN
          Return v_unickname;
   
   ELSIF  p_code = 3          THEN
          Return v_uatid;
          
   ELSE   
          Return 'p_code 오류';
   END IF;
       
           
End;
